170:       # Determine the alias used for remote_table_name, if any. In all cases this will already
171:       # have been assigned an ID in table_ids (either through being involved in a previous join,
172:       # or - if it's the first table in the query - as the default value of table_ids)
173:       remote_table_alias = remote_table_name = association_class.table_name
174:       remote_table_alias += "_#{table_ids[remote_table_name]}" unless table_ids[remote_table_name] == 1
175: 
176:       # Assign a new alias for the local table.
177:       local_table_alias = local_table_name = reflection.active_record.table_name
178:       if table_ids[local_table_name]
179:         table_id = table_ids[local_table_name] += 1
180:         local_table_alias += "_#{table_id}"
181:       else
182:         table_ids[local_table_name] = 1
183:       end
184:       
185:       conditions = ''
186:       # Add filter for single-table inheritance, if applicable.
187:       conditions += " AND #{remote_table_alias}.#{association_class.inheritance_column} = #{association_class.quote_value(association_class.name.demodulize)}" unless association_class.descends_from_active_record?
188:       # Add custom conditions
189:       conditions += " AND (#{interpolate_sql(association_class.send(:sanitize_sql, reflection.options[:conditions]))})" if reflection.options[:conditions]
190:       
191:       if reflection.macro == :belongs_to
192:         if reflection.options[:polymorphic]
193:           conditions += " AND #{local_table_alias}.#{reflection.options[:foreign_type]} = #{reflection.active_record.quote_value(association_class.base_class.name.to_s)}"
194:         end
195:         {
196:           :joins => reflection.options[:joins],
197:           :remote_key => "#{remote_table_alias}.#{association_class.primary_key}",
198:           :local_key => reflection.primary_key_name,
199:           :conditions => conditions
200:         }
201:       else
202:         # Association is has_many (without :through)
203:         if reflection.options[:as]

131:       # Construct the join components of the source association, so that we have a path from
132:       # the eventual target table of the association up to the table named in :through, and
133:       # all tables involved are allocated table IDs.
134:       source_attrs = construct_nested_join_attributes(reflection.source_reflection, reflection.klass, table_ids)
135:       
136:       # Determine the alias of the :through table; this will be the last table assigned
137:       # when constructing the source join components above.
138:       through_table_alias = through_table_name = reflection.through_reflection.table_name
139:       through_table_alias += "_#{table_ids[through_table_name]}" unless table_ids[through_table_name] == 1
140: 
141:       # Construct the join components of the through association, so that we have a path to
142:       # the active record's table.
143:       through_attrs = construct_nested_join_attributes(reflection.through_reflection, reflection.through_reflection.klass, table_ids)
144: 
145:       # Any subsequent joins / filters on owner attributes will act on the through association,
146:       # so that's what we return for the conditions/keys of the overall association.
147:       conditions = through_attrs[:conditions]

construct_nested_join_attributes( reflection = @reflection, association_class = reflection.klass, table_ids = {association_class.table_name => 1})

Given any belongs_to or has_many (including has_many :through) association, return the essential components of a join corresponding to that association, namely:

  • :joins: any additional joins required to get from the association‘s table (reflection.table_name) to the table that‘s actually joining to the active record‘s table
  • :remote_key: the name of the key in the join table (qualified by table name) which will join to a field of the active record‘s table
  • :local_key: the name of the key in the local table (not qualified by table name) which will take part in the join
  • :conditions: any additional conditions (e.g. filtering by type for a polymorphic association, 
     or a :conditions clause explicitly given in the association), including a leading AND
